**详细任务说明**:

#### P3-001 实现Trace来源追踪机制

```rust
// Trace - 来源追踪信息
struct Trace {
    // 来源标识
    source_type: SourceType,       // Storage 或 Api
    source_id: String,             // 用户自定义标识
    source_name: String,           // 显示名称
    data_type: TraceDataType,      // Track/Artist/Album/Playlist
    
    // API源配置
    base_url: Option<String>,      // API基础地址
    
    // 数据标识
    data_id: String,               // 数据在该来源中的ID
    fetch_method: FetchMethod,     // LocalFile/Download/ApiCall
}

enum SourceType {
    Storage(StorageType),  // Local 或 WebDAV
    Api,                   // API型（不硬编码具体平台）
}
```

#### P3-002 实现ApiSource抽象类

```javascript
// src/api/source/api.js
class ApiSource extends Source {
    constructor(sourceId, sourceName, baseUrl) {
        super(sourceId, sourceName, 'api', SourceType.API);
        this.baseUrl = baseUrl;
    }
    
    // 搜索功能
    async searchTracks(keyword, options = {}) {
        const response = await fetch(
            `${this.baseUrl}/search?q=${encodeURIComponent(keyword)}&type=tracks`
        );
        return response.json();
    }
    
    // 创建Trace
    createTrace(dataType, dataId, fetchMethod) {
        return Trace.createApiSource(
            this.sourceId,
            this.sourceName,
            this.baseUrl,
            dataType,
            dataId,
            fetchMethod
        );
    }
}
```

#### P3-003 实现SourceManager多源管理

```javascript
// src/api/source/index.js
class SourceManager {
    #sources = new Map();
    
    registerSource(sourceId, source) {
        this.#sources.set(sourceId, source);
    }
    
    getSource(sourceId) {
        return this.#sources.get(sourceId);
    }
    
    // 根据Trace跳转到来源数据
    async navigateByTrace(trace) {
        const source = this.getSource(trace.sourceId);
        if (!source) {
            throw new Error(`Source not found: ${trace.sourceId}`);
        }
        return await source.getByTrace(trace);
    }
}
```
