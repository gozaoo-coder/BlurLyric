```javascript
    #id;

    #refCount;
    #onDestroy;

    release() { 
        this.#refCount--;
        if (this.#refCount <= 0) this.destroy();

        }
    }
}

class ResourcePool {

            this.#evict(); // LRU淘汰
        }
        this.#resources.set(key, resource);
    }
}
```

```rust
// 生成歌曲指纹
fn generate_fingerprint(song: &Song) -> String {
    let normalized_title = normalize_for_dedup(&song.name);
    let normalized_artists: Vec<String> = song.ar.iter()
        .map(|a| normalize_for_dedup(&a.name))
        .collect();
    let normalized_album = normalize_for_dedup(&song.al.name);
    
    format!("{}|{}|{}", 
        normalized_title,
        normalized_artists.join("&"),
        normalized_album
    )
}

// 计算音质评分
fn calculate_song_quality_score(song: &Song) -> u32 {
    let mut score = 0u32;
    
    // 比特率分数 (最高320分)
    if let Some(bitrate) = song.bitrate {
        score += bitrate.min(320);
    }
    
    // 格式分数 (无损格式优先)
    score += match format.to_lowercase().as_str() {
        "flac" => 500,
        "wav" | "aiff" => 400,
        "aac" | "m4a" => 300,
        "mp3" => 200,
        "ogg" => 250,
        _ => 100,
    };
    
    // 采样率分数 (最高480分)
    if let Some(sample_rate) = song.sample_rate {
        score += (sample_rate / 100).min(480);
    }
    
    score
}
```

### 3.6.4 去重策略

| 策略 | 描述 | 权重 |
|------|------|------|
| 格式优先 | FLAC > WAV > AAC > MP3 | 500分 |
| 比特率优先 | 更高比特率优先 | 320分 |
| 采样率优先 | 更高采样率优先 | 480分 |
| 时长优先 | 完整曲目优先 | 100分 |

```javascript
/**
 * Track - 单曲数据模型
 * 支持多来源追踪和统一资源获取

class Track {
    #data;
    #albumCover;
    #musicFile;
    
    // ========== 基本属性 ==========
    get id() { return this.#data.id; }
    get name() { return this.#data.name; }
    get artists() { 
        return (this.#data.ar ?? []).map(ar => ArtistSummup.fromRaw(ar)); 
    }
    get album() { 
        return this.#data.al ? AlbumSummup.fromRaw(this.#data.al) : null; 
    }
    get duration() { return this.#data.duration ?? null; }
    
    // ========== 来源追踪 ==========
    get traces() { 
        return (this.#data.traces ?? []).map(t => Trace.fromRaw(t)); 
    }
    get primaryTrace() { 
        const traces = this.traces;
        return traces[this.#data.primary_trace_index ?? 0] ?? null;
    }
    hasLocalResource() {
        return this.#data.local_resource != null;
    }
    
    // ========== 用户偏好状态 ==========
    get isLiked() { return this.#data.is_liked ?? false; }
    get playlistIds() { return this.#data.playlist_ids ?? []; }
    
    // ========== 资源获取（遵守资源调用流程）==========
    
    /**
     * 获取音频文件
     * 流程：Trace.fetchMethod → IPC → 缓存检查 → 下载/读取 → ObjectURL
     */
    async getMusicFile(resourceFetcher) {
        return await resourceFetcher.getAudioFile(this);
    }
    
    /**
     * 获取专辑封面
     */
    async getAlbumCover(resourceFetcher, resolution = 368) {
        if (this.album) {
            return await resourceFetcher.getAlbumCover(this.album, resolution);
        }
        return null;
    }
    
    /**
     * 跳转到艺人页面
     */
    navigateToArtist(artistIndex) {
        const artist = this.artists[artistIndex];
        if (artist) {
            return artist.navigateToSource();
        }
        return null;
    }
    
    /**
     * 跳转到专辑页面
     */
    navigateToAlbum() {
        if (this.album) {
            return this.album.navigateToSource();
        }
        return null;
    }
    
    releaseResources() {
        this.#albumCover?.destroy?.();
        this.#musicFile?.destroy?.();
    }
}
```
