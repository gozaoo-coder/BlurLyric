### 5.2.2 TauriSource 实现

**文件位置**: `src/api/source/tauri.js`

```javascript
import { invoke } from '@tauri-apps/api/core';
import { Source } from './base.js';

/**
 * TauriSource - 本地数据源实现
 * 通过Tauri IPC与Rust后端通信
 */
class TauriSource extends Source {
    constructor() {
        super('本地音乐库', 'local');
    }

    // ========== 音乐列表操作 ==========
    
    async getMusicList() {
        return await invoke('get_music_list');
    }

    async getAllMusicDirs() {
        return await invoke('get_all_music_dirs');
    }

    async addMusicDirs(dirs) {
        const dirArray = Array.isArray(dirs) ? dirs : [dirs];
        return await invoke('add_music_dirs', { newDirs: dirArray });
    }

    async removeMusicDirs(dirs) {
        const dirArray = Array.isArray(dirs) ? dirs : [dirs];
        return await invoke('remove_music_dirs', { dirsToRemove: dirArray });
    }

    async refreshMusicCache() {
        return await invoke('refresh_music_cache');
    }

    // ========== 专辑操作 ==========

    async getAlbums() {
        return await invoke('get_all_my_albums');
    }

    async getAlbumById(albumId) {
        return await invoke('get_album_by_id', { albumId });
    }

    async getAlbumsSongsById(albumId) {
        return await invoke('get_albums_songs_by_id', { albumId });
    }

    // ========== 艺术家操作 ==========

    async getArtists() {
        return await invoke('get_all_my_artists');
    }

    async getArtistById(artistId) {
        return await invoke('get_artist_by_id', { artistId });
    }

    async getArtistsSongsById(artistId) {
        return await invoke('get_artists_songs_by_id', { artistId });
    }

    // ========== 资源获取 ==========

    async getAlbumCover(albumId, maxResolution = 368) {
        const response = await invoke('get_low_quality_album_cover', { 
            albumId, 
            maxResolution 
        });
        const blob = new Blob([response], { type: 'image/webp' });
        const objectURL = URL.createObjectURL(blob);
        return {
            objectURL,
            destroyObjectURL: () => URL.revokeObjectURL(objectURL)
        };
    }

    async getMusicFile(songId) {
        const response = await invoke('get_music_file', { songId });
        const blob = new Blob([response]);
        const objectURL = URL.createObjectURL(blob);
        return {
            objectURL,
            destroyObjectURL: () => URL.revokeObjectURL(objectURL)
        };
    }

    // ========== 搜索功能 ==========

    async searchTracks(keyword, options = {}) {
        const allTracks = await this.getMusicList();
        const lowerKeyword = keyword.toLowerCase();
        return allTracks.filter(track => 
            track.name.toLowerCase().includes(lowerKeyword) ||
            track.ar?.some(a => a.name.toLowerCase().includes(lowerKeyword))
        );
    }

    async searchAlbums(keyword, options = {}) {
        const albums = await this.getAlbums();
        const lowerKeyword = keyword.toLowerCase();
        return albums.filter(album => 
            album.name.toLowerCase().includes(lowerKeyword)
        );
    }

    async searchArtists(keyword, options = {}) {
        const artists = await this.getArtists();
        const lowerKeyword = keyword.toLowerCase();
        return artists.filter(artist => 
            artist.name.toLowerCase().includes(lowerKeyword)
        );
    }

    async searchAll(keyword, options = {}) {
        const [tracks, albums, artists] = await Promise.all([
            this.searchTracks(keyword, options),
            this.searchAlbums(keyword, options),
            this.searchArtists(keyword, options)
        ]);
        return { tracks, albums, artists };
    }

    // ========== 初始化 ==========

    async initApplication() {
        return await invoke('init_application');
    }
}
```

### 5.2.3 Resource 类

**文件位置**: `src/api/resources/index.js`

```javascript
/**
 * Resource - 资源基类
 * 实现引用计数和自动释放
 */
class Resource {
    #id;
    #objectURL;
    #refCount;
    #onDestroy;

    constructor(id, objectURL, onDestroy = null) {
        this.#id = id;
        this.#objectURL = objectURL;
        this.#refCount = 1;
        this.#onDestroy = onDestroy;
    }

    get objectURL() { return this.#objectURL; }
    get id() { return this.#id; }
    get refCount() { return this.#refCount; }

    retain() {
        this.#refCount++;
        return this;
    }

    release() {
        this.#refCount--;
        if (this.#refCount <= 0) {
            this.destroy();
            return true;
        }
        return false;
    }

    destroy() {
        if (this.#objectURL && this.#objectURL.startsWith('blob:')) {
            URL.revokeObjectURL(this.#objectURL);
        }
        this.#onDestroy?.();
        this.#objectURL = null;
        this.#onDestroy = null;
    }
}

/**
 * ResourcePool - 资源池
 * 实现LRU淘汰策略
 */
class ResourcePool {
    #resources = new Map();
    #maxSize;

    constructor(maxSize = 100) {
        this.#maxSize = maxSize;
    }

    get(key) { return this.#resources.get(key); }
    has(key) { return this.#resources.has(key); }
    get size() { return this.#resources.size; }

    set(key, resource) {
        if (this.#resources.size >= this.#maxSize) {
            this.#evict();
        }
        this.#resources.set(key, resource);
    }

    delete(key) {
        const resource = this.#resources.get(key);
        if (resource) {
            resource.destroy();
            this.#resources.delete(key);
        }
    }

    clear() {
        this.#resources.forEach(resource => resource.destroy());
        this.#resources.clear();
    }

    #evict() {
        const entries = [...this.#resources.entries()];
        if (entries.length > 0) {
            const [firstKey] = entries[0];
            this.delete(firstKey);
        }
    }
}
```

```javascript
/**
 * Track - 单曲数据模型
 */
class Track {
    #data;
    #albumCover;
    #musicFile;

    constructor(trackData) {
        this.#data = trackData;
        this.#albumCover = null;
        this.#musicFile = null;
    }

    static fromRaw(rawData) {
        return new Track(rawData);
    }

    static fromRawArray(rawArray) {
        return rawArray.map(raw => Track.fromRaw(raw));
    }

    // ========== 基本属性 ==========
    
    get id() { return this.#data.id; }
    get name() { return this.#data.name; }
    get src() { return this.#data.src; }
    get lyric() { return this.#data.lyric; }
    get trackNumber() { return this.#data.track_number ?? this.#data.trackNumber ?? 0; }
    
    get artists() {
        return (this.#data.ar ?? []).map(ar => Artist.fromRaw(ar));
    }
    
    get album() {
        return this.#data.al ? Album.fromRaw(this.#data.al) : null;
    }

    // ========== 扩展属性 ==========
    
    get duration() { return this.#data.duration ?? null; }
    
    get durationFormatted() {
        const duration = this.#data.duration;
        if (!duration || duration <= 0) return '--:--';
        const minutes = Math.floor(duration / 60);
        const seconds = Math.floor(duration % 60);
        return `${minutes}:${seconds.toString().padStart(2, '0')}`;
    }
    
    get genre() { return this.#data.genre ?? null; }
    get year() { return this.#data.year ?? null; }
    get bitrate() { return this.#data.bitrate ?? null; }
    get sampleRate() { return this.#data.sampleRate ?? this.#data.sample_rate ?? null; }

    // ========== 多来源属性 ==========
    
    get sources() { return this.#data.sources ?? []; }
    get primarySourceIndex() { return this.#data.primarySourceIndex ?? 0; }
    get sourceCount() { return this.#data.sourceCount ?? this.sources.length ?? 1; }
    
    get primarySource() {
        const sources = this.sources;
        const index = this.primarySourceIndex;
        return sources[index] ?? sources[0] ?? null;
    }
    
    get alternativeSources() {
        const sources = this.sources;
        const primaryIndex = this.primarySourceIndex;
        return sources.filter((_, index) => index !== primaryIndex);
    }
    
    hasMultipleSources() {
        return this.sourceCount > 1;
    }

    // ========== 资源获取 ==========
    
    async getAlbumCover(apiAdapter, resolution = 368) {
        if (!this.#albumCover) {
            const albumId = this.#data.al?.id ?? -1;
            if (albumId < 0) return null;
            this.#albumCover = new AlbumCoverResource(albumId, resolution);
        }
        return this.#albumCover.load(apiAdapter);
    }

    async getMusicFile(apiAdapter) {
        if (!this.#musicFile) {
            this.#musicFile = new MusicFileResource(this.#data.id);
        }
        return this.#musicFile.load(apiAdapter);
    }

    releaseResources() {
        if (this.#albumCover) {
            this.#albumCover.destroy();
            this.#albumCover = null;
        }
        if (this.#musicFile) {
            this.#musicFile.destroy();
            this.#musicFile = null;
        }
    }

    toRaw() {
        return { ...this.#data };
    }
}
```

## 5.5 Trace 来源追踪机制

### 5.5.1 Trace 设计理念

Trace 是所有数据模型（Track、Artist、Album）的核心追踪机制，实现以下目标：

| 目标 | 说明 |
|------|------|
| 来源追踪 | 记录数据来自哪个 Source |
| 类型标识 | 标明数据类型（track/artist/album） |
| 资源获取 | 提供获取资源的统一方法 |
| 精准跳转 | 根据来源和ID实现精准跳转 |

### 5.5.2 Trace 数据结构

```rust
/// 数据类型枚举
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub enum TraceDataType {
    Track,      // 单曲
    Artist,     // 艺人
    Album,      // 专辑
    Playlist,   // 歌单
}

/// 来源类型枚举
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub enum SourceType {
    Storage(StorageType),   // 存储型
    Api,                    // API型（不硬编码具体平台）
}

/// 存储型来源
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub enum StorageType {
    Local,      // 本地文件 (TauriSource)
    WebDAV,     // WebDAV存储
}

/// Trace - 来源追踪信息
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Trace {
    // ========== 来源标识 ==========
    /// 来源类型
    pub source_type: SourceType,
    /// 来源唯一标识（用户自定义，如 "my-music-api-1"）
    pub source_id: String,
    /// 来源名称（显示用，如 "我的音乐API"）
    pub source_name: String,
    /// 数据类型
    pub data_type: TraceDataType,
    
    // ========== API源配置（仅Api类型需要）==========
    /// API基础URL（符合接口规范即可接入）
    pub base_url: Option<String>,
    
    // ========== 数据标识 ==========
    /// 数据在该来源中的唯一ID
    pub data_id: String,
    /// 数据在该来源中的URL（可选）
    pub data_url: Option<String>,
    
    // ========== 资源信息 ==========
    /// 资源获取方法
    pub fetch_method: FetchMethod,
    /// 资源元信息
    pub resource_info: Option<ResourceInfo>,
    
    // ========== 扩展元数据 ==========
    /// 平台特有元数据（通用KV结构）
    pub metadata: HashMap<String, serde_json::Value>,
}

/// 资源获取方法
#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum FetchMethod {
    /// 本地文件：直接读取
    LocalFile { path: PathBuf },
    /// 需要下载：从URL下载到本地
    Download { url: String, format: String },
    /// 流式播放：直接流式访问（不推荐）
    Stream { url: String },
    /// 需要API调用：通过API获取
    ApiCall { endpoint: String, params: HashMap<String, String> },
}

/// 资源元信息
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ResourceInfo {
    pub format: Option<String>,        // 格式 (mp3/flac等)
    pub bitrate: Option<u32>,          // 比特率
    pub sample_rate: Option<u32>,      // 采样率
    pub size: Option<u64>,             // 文件大小
    pub duration: Option<f64>,         // 时长
    pub quality_score: Option<u32>,    // 音质评分
}

impl Trace {
    /// 创建本地文件Trace
    pub fn local_file(
        path: PathBuf, 
        data_type: TraceDataType, 
        data_id: String
    ) -> Self {
        Trace {
            source_type: SourceType::Storage(StorageType::Local),
            source_id: "local".to_string(),
            source_name: "本地音乐库".to_string(),
            data_type,
            base_url: None,
            data_id,
            data_url: None,
            fetch_method: FetchMethod::LocalFile { path },
            resource_info: None,
            metadata: HashMap::new(),
        }
    }
    
    /// 创建WebDAV Trace
    pub fn webdav(
        source_id: String,
        source_name: String,
        data_type: TraceDataType,
        data_id: String,
        path: String,
    ) -> Self {
        Trace {
            source_type: SourceType::Storage(StorageType::WebDAV),
            source_id,
            source_name,
            data_type,
            base_url: None,
            data_id,
            data_url: Some(path),
            fetch_method: FetchMethod::LocalFile { path: PathBuf::new() }, // WebDAV路径
            resource_info: None,
            metadata: HashMap::new(),
        }
    }
    
    /// 创建API来源Trace
    pub fn api_source(
        source_id: String,
        source_name: String,
        base_url: String,
        data_type: TraceDataType,
        data_id: String,
        fetch_method: FetchMethod,
    ) -> Self {
        Trace {
            source_type: SourceType::Api,
            source_id,
            source_name,
            data_type,
            base_url: Some(base_url),
            data_id,
            data_url: None,
            fetch_method,
            resource_info: None,
            metadata: HashMap::new(),
        }
    }
    
    /// 检查是否为存储型来源
    pub fn is_storage(&self) -> bool {
        matches!(self.source_type, SourceType::Storage(_))
    }
    
    /// 检查是否为API型来源
    pub fn is_api(&self) -> bool {
        matches!(self.source_type, SourceType::Api)
    }
    
    /// 检查是否为本地文件
    pub fn is_local(&self) -> bool {
        matches!(self.source_type, SourceType::Storage(StorageType::Local))
    }
    
    /// 检查是否为WebDAV
    pub fn is_webdav(&self) -> bool {
        matches!(self.source_type, SourceType::Storage(StorageType::WebDAV))
    }
    
    /// 获取资源（根据Trace类型）
    pub async fn fetch_resource(&self) -> Result<Vec<u8>, String> {
        match &self.fetch_method {
            FetchMethod::LocalFile { path } => {
                tokio::fs::read(path).await.map_err(|e| e.to_string())
            }
            FetchMethod::Download { url, format } => {
                download_to_temp_cache(url, format).await
            }
            FetchMethod::Stream { url } => {
                fetch_stream(url).await
            }
            FetchMethod::ApiCall { endpoint, params } => {
                fetch_via_api(endpoint, params).await
            }
        }
    }
    
    /// 跳转到来源数据
    /// 根据source_id和data_id精准跳转
    pub async fn navigate_to_source(&self) -> Result<Box<dyn BaseModel>, String> {
        // 通过SourceManager获取对应的Source
        let source = SourceManager::get_source(&self.source_id)?;
        
        // 根据数据类型调用对应的获取方法
        match self.data_type {
            TraceDataType::Track => {
                let track = source.get_track(&self.data_id).await?;
                Ok(Box::new(track))
            }
            TraceDataType::Artist => {
                let artist = source.get_artist(&self.data_id).await?;
                Ok(Box::new(artist))
            }
            TraceDataType::Album => {
                let album = source.get_album(&self.data_id).await?;
                Ok(Box::new(album))
            }
            TraceDataType::Playlist => {
                let playlist = source.get_playlist(&self.data_id).await?;
                Ok(Box::new(playlist))
            }
        }
    }
}
```

```javascript
/**
 * Trace - 来源追踪信息（前端）
 * 
 * 设计原则：
 * 1. 不硬编码任何具体音乐平台名称
 * 2. 通过 source_id 标识用户配置的数据源
 * 3. 通过 base_url 作为 API 源配置
 * 4. 符合接口规范即可接入
 */
class Trace {
    #data;
    
    constructor(traceData) {
        this.#data = traceData;
    }
    
    // ========== 基本属性 ==========
    
    get sourceType() { return this.#data.sourceType; }
    get sourceId() { return this.#data.sourceId; }
    get sourceName() { return this.#data.sourceName; }
    get dataType() { return this.#data.dataType; }
    get dataId() { return this.#data.dataId; }
    get baseUrl() { return this.#data.baseUrl; }
    
    // ========== 类型检查 ==========
    
    isStorage() {
        return this.#data.sourceType?.type === 'storage';
    }
    
    isApi() {
        return this.#data.sourceType?.type === 'api';
    }
    
    isLocal() {
        return this.#data.sourceType?.storage === 'local';
    }
    
    isWebDAV() {
        return this.#data.sourceType?.storage === 'webdav';
    }
    
    // ========== 资源获取 ==========
    
    /**
     * 获取资源（根据Trace类型）
     * - Track: 获取音频文件
     * - Album: 获取专辑封面
     * - Artist: 获取艺人头像
     */
    async fetchResource() {
        const method = this.#data.fetchMethod;
        
        switch (method.type) {
            case 'LocalFile':
                return await this.fetchLocalFile(method.path);
            case 'Download':
                return await this.downloadResource(method.url, method.format);
            case 'ApiCall':
                return await this.fetchViaApi(method.endpoint, method.params);
            default:
                throw new Error(`Unknown fetch method: ${method.type}`);
        }
    }
    
    /**
     * 获取本地文件
     */
    async fetchLocalFile(path) {
        const result = await invoke('get_local_file', { path });
        return {
            objectURL: result.objectURL,
            destroy: result.destroyObjectURL
        };
    }
    
    /**
     * 下载网络资源
     */
    async downloadResource(url, format) {
        const result = await invoke('download_resource_to_temp', {
            url,
            format,
            traceId: this.dataId
        });
        return {
            objectURL: result.objectURL,
            destroy: result.destroyObjectURL
        };
    }
    
    /**
     * 通过API获取资源
     */
    async fetchViaApi(endpoint, params) {
        const result = await invoke('fetch_via_api', {
            baseUrl: this.baseUrl,
            endpoint,
            params
        });
        return {
            objectURL: result.objectURL,
            destroy: result.destroyObjectURL
        };
    }
    
    /**
     * 跳转到来源数据
     * 根据 source_id 和 data_id 精准跳转
     */
    async navigateToSource() {
        const { sourceManager } = await import('./source/index.js');
        const source = sourceManager.getSource(this.sourceId);
        
        if (!source) {
            throw new Error(`Source not found: ${this.sourceId}`);
        }
        
        switch (this.#data.dataType) {
            case 'track':
                return await source.getTrackDetail(this.dataId);
            case 'artist':
                return await source.getArtistDetail(this.dataId);
            case 'album':
                return await source.getAlbumDetail(this.dataId);
            case 'playlist':
                return await source.getPlaylistDetail(this.dataId);
            default:
                throw new Error(`Unknown data type: ${this.#data.dataType}`);
        }
    }
    
    toRaw() {
        return { ...this.#data };
    }
    
    static fromRaw(data) {
        return new Trace(data);
    }
    
    /**
     * 创建本地文件Trace
     */
    static createLocalFile(path, dataType, dataId) {
        return new Trace({
            sourceType: { type: 'storage', storage: 'local' },
            sourceId: 'local',
            sourceName: '本地音乐库',
            dataType,
            baseUrl: null,
            dataId,
            dataUrl: null,
            fetchMethod: { type: 'LocalFile', path },
            resourceInfo: null,
            metadata: {}
        });
    }
    
    /**
     * 创建API来源Trace
     */
    static createApiSource(sourceId, sourceName, baseUrl, dataType, dataId, fetchMethod) {
        return new Trace({
            sourceType: { type: 'api' },
            sourceId,
            sourceName,
            dataType,
            baseUrl,
            dataId,
            dataUrl: null,
            fetchMethod,
            resourceInfo: null,
            metadata: {}
        });
    }
}
```

## 5.6 统一数据模型

### 5.6.1 数据模型设计原则

| 原则 | 说明 |
|------|------|
| 统一接口 | 所有数据模型实现相同的特征接口 |
| Trace追踪 | 所有模型都有 `traces: Vec<Trace>` |
| 来源聚合 | 同一首歌可能来自多个来源，通过Trace追踪 |
| 统一获取 | 通过Trace获取资源，不直接依赖具体Source |

### 5.6.2 BaseModel 基础特征

```rust
/// 数据模型基础特征
pub trait BaseModel: Serialize + for<'de> Deserialize<'de> + Clone + Send + Sync {
    /// 获取唯一ID
    fn id(&self) -> &str;
    
    /// 获取名称
    fn name(&self) -> &str;
    
    /// 获取所有Trace
    fn traces(&self) -> &[Trace];
    
    /// 添加Trace
    fn add_trace(&mut self, trace: Trace);
    
    /// 获取主Trace
    fn primary_trace(&self) -> Option<&Trace> {
        self.traces().first()
    }
    
    /// 根据来源类型获取Trace
    fn get_trace_by_source(&self, source_type: &SourceType) -> Option<&Trace> {
        self.traces().iter()
            .find(|t| &t.source_type == source_type)
    }
    
    /// 获取本地资源Trace（优先）
    fn get_local_trace(&self) -> Option<&Trace> {
        self.traces().iter()
            .find(|t| t.is_storage())
    }
    
    /// 检查是否有本地资源
    fn has_local_resource(&self) -> bool {
        self.traces().iter().any(|t| t.is_storage())
    }
}
```

## 5.6 ApiSource 网络源规范

### 5.6.1 设计原则

| 原则 | 说明 |
|------|------|
| 按需获取 | 不全量同步，通过搜索/歌单获取Track信息 |
| 来源追踪 | Track包含完整的来源信息（平台、ID、获取方法） |
| 本地优先 | 资源必须下载到本地后才可播放 |
| 缓存分层 | 临时缓存池 vs 用户偏好池 |

```javascript
/**
 * NeteaseMusicSource - 网易云音乐数据源
 */
class NeteaseMusicSource extends ApiSource {
    constructor(config) {
        super('网易云音乐', 'netease', config);
    }
    
    async searchTracks(keyword, options = {}) {
        const response = await fetch(
            `${this.baseUrl}/search?keywords=${encodeURIComponent(keyword)}&type=1`
        );
        const data = await response.json();
        
        return {
            tracks: data.result.songs.map(song => this.parseTrack(song)),
            total: data.result.songCount,
            hasMore: data.result.hasMore
        };
    }
    
    parseTrack(song) {
        return {
            // 基本信息
            id: `netease_${song.id}`,  // 本地唯一ID
            name: song.name,
            artists: song.ar.map(a => ({ id: a.id, name: a.name })),
            album: {
                id: song.al.id,
                name: song.al.name,
                picUrl: song.al.picUrl
            },
            duration: song.dt / 1000,
            
            // 来源信息（关键）
            sources: [{
                sourceType: 'netease',
                sourceId: song.id.toString(),
                sourceName: '网易云音乐',
                resourceUrl: null,  // 按需获取
                resourceInfo: {
                    format: 'mp3',
                    bitrate: song.h?.br || 320,
                    size: song.h?.size || 0,
                    qualityScore: this.calculateQuality(song.h?.br || 320)
                },
                fetchMethod: 'download',  // 需要下载
                metadata: {
                    platform: 'netease',
                    fee: song.fee,        // 是否需要VIP
                    copyright: song.copyright
                }
            }],
            
            primarySourceIndex: 0,
            localResource: null,  // 尚未下载
            isLiked: false,
            playlistIds: []
        };
    }
    
    async getTrackResourceUrl(trackId, options = {}) {
        const response = await fetch(
            `${this.baseUrl}/song/url?id=${trackId}&br=${options.bitrate || 320000}`
        );
        const data = await response.json();
        
        return {
            url: data.data[0].url,
            format: 'mp3',
            bitrate: data.data[0].br,
            size: data.data[0].size,
            expiresAt: Date.now() + 3600000  // 1小时后过期
        };
    }
```

### 5.8.4 前端资源调用实现

```javascript
/**
 * ResourceFetcher - 统一资源获取器
 * 
 * 实现完整的资源调用流程
 */
class ResourceFetcher {
    // ========== 缓存层 ==========
    
    // ObjectURL缓存（对应 tauri.js 中的实现）
    #objectURLCache = new Map();      // id → objectURL
    #objectURLCounter = new Map();    // id → 引用计数
    #requestCache = new Map();        // 请求去重
    
    // ========== 资源获取入口 ==========
    
    /**
     * 获取大文件资源（完整流程）
     * @param {Trace} trace 来源追踪信息
     * @param {string} resourceId 资源ID
     * @returns {Promise<{objectURL: string, destroyObjectURL: Function}>}
     */
    async fetchLargeResource(trace, resourceId) {
        // 步骤1-3: 根据Trace获取资源
        const cacheKey = this.generateCacheKey(trace, resourceId);
        
        // 步骤8: 检查前端ObjectURL缓存
        if (this.#objectURLCache.has(cacheKey)) {
            this.#objectURLCounter.set(
                cacheKey, 
                this.#objectURLCounter.get(cacheKey) + 1
            );
            return {
                objectURL: this.#objectURLCache.get(cacheKey),
                destroyObjectURL: () => this.destroyObjectURL(cacheKey)
            };
        }
        
        // 请求去重
        if (this.#requestCache.has(cacheKey)) {
            return this.#requestCache.get(cacheKey);
        }
        
        // 步骤3: Tauri IPC调用
        const requestPromise = this.fetchViaIPC(trace, resourceId, cacheKey);
        this.#requestCache.set(cacheKey, requestPromise);
        
        return requestPromise;
    }
    
    /**
     * 通过IPC获取资源
     */
    async fetchViaIPC(trace, resourceId, cacheKey) {
        try {
            // 步骤3: Tauri IPC
            const data = await invoke('fetch_resource', {
                sourceType: trace.sourceType,
                sourceId: trace.sourceId,
                dataType: trace.dataType,
                dataId: trace.dataId,
                fetchMethod: trace.fetchMethod,
                resourceId
            });
            
            // 步骤7: IPC返回二进制数据
            // 步骤8: 创建ObjectURL并存入缓存
            const blob = new Blob([data]);
            const objectURL = URL.createObjectURL(blob);
            
            this.#objectURLCache.set(cacheKey, objectURL);
            this.#objectURLCounter.set(cacheKey, 1);
            this.#requestCache.delete(cacheKey);
            
            // 步骤9: 返回ObjectURL和释放器
            return {
                objectURL,
                destroyObjectURL: () => this.destroyObjectURL(cacheKey)
            };
        } catch (error) {
            this.#requestCache.delete(cacheKey);
            throw error;
        }
    }
    
    /**
     * 生成缓存Key
     */
    generateCacheKey(trace, resourceId) {
        const prefix = {
            track: 'mf',      // music file
            album: 'al',      // album cover
            artist: 'ar'      // artist avatar
        }[trace.dataType] || 'res';
        
        return `${prefix}_${trace.sourceId}_${resourceId}`;
    }
    
    /**
     * 销毁ObjectURL（引用计数管理）
     */
    destroyObjectURL(cacheKey) {
        if (!this.#objectURLCache.has(cacheKey)) return;
        
        const count = this.#objectURLCounter.get(cacheKey) - 1;
        this.#objectURLCounter.set(cacheKey, count);
        
        if (count <= 0) {
            // 引用计数为0，释放资源
            URL.revokeObjectURL(this.#objectURLCache.get(cacheKey));
            this.#objectURLCache.delete(cacheKey);
            this.#objectURLCounter.delete(cacheKey);
        }
    }
    
    // ========== 便捷方法 ==========
    
    /**
     * 获取音频文件
     */
    async getAudioFile(track) {
        const trace = track.primaryTrace();
        return this.fetchLargeResource(trace, track.id);
    }
    
    /**
     * 获取专辑封面
     */
    async getAlbumCover(album, resolution = 368) {
        const trace = album.primaryTrace();
        return this.fetchLargeResource(trace, `${album.id}_${resolution}`);
    }
    
    /**
     * 获取艺人头像
     */
    async getArtistAvatar(artist, resolution = 368) {
        const trace = artist.primaryTrace();
        return this.fetchLargeResource(trace, `${artist.id}_${resolution}`);
    }
    
    /**
     * 获取文本内容（不经过缓存流程）
     */
    async fetchTextContent(trace) {
        return await trace.navigateToSource();
    }
}

// 单例导出

```

### 5.8.5 Rust后端资源获取实现

```rust
/// 资源获取命令
#[tauri::command]
async fn fetch_resource(
    source_type: SourceType,
    source_id: String,
    data_type: TraceDataType,
    data_id: String,
    fetch_method: FetchMethod,
    resource_id: String,
) -> Result<Vec<u8>, String> {
    // 步骤4: 根据Source类型判断获取方式
    match source_type {
        // 存储型：直接读取本地文件
        SourceType::Storage(storage_type) => {
            match storage_type {
                StorageType::Local => {
                    // 本地文件，直接读取
                    if let FetchMethod::LocalFile { path } = fetch_method {
                        return tokio::fs::read(&path)
                            .await
                            .map_err(|e| e.to_string());
                    }
                    Err("Invalid fetch method for local storage".to_string())
                }
                StorageType::WebDAV => {
                    // WebDAV存储，检查缓存后获取
                    fetch_from_webdav_or_cache(source_id, data_id, fetch_method).await
                }
            }
        }
        
        // API型：检查缓存架构
        SourceType::Api => {
            // 步骤5-6: 检查缓存，无则获取并缓存
            fetch_from_api_or_cache(source_id, data_id, fetch_method, data_type).await
        }
    }
}

/// 从API获取或从缓存读取
async fn fetch_from_api_or_cache(
    source_id: String,
    data_id: String,
    fetch_method: FetchMethod,
    data_type: TraceDataType,
) -> Result<Vec<u8>, String> {
    let cache_manager = CacheManager::get_instance()?;
    
    // 检查缓存
    if let Some(cached) = cache_manager.get_cached(&data_id)? {
        // 已缓存，直接返回
        return tokio::fs::read(&cached.local_path)
            .await
            .map_err(|e| e.to_string());
    }
    
    // 未缓存，根据fetch_method获取
    let data = match fetch_method {
        FetchMethod::Download { url, .. } => {
            download_from_url(&url).await?
        }
        FetchMethod::ApiCall { endpoint, params } => {
            call_api_endpoint(&endpoint, params).await?
        }
        _ => return Err("Invalid fetch method for API source".to_string()),
    };
    
    // 步骤6: 存入缓存分层架构
    // 检查用户偏好状态决定缓存类型
    let user_pref = UserPreference::get_instance()?;
    let cache_type = if user_pref.is_liked(&data_id) || user_pref.in_playlist(&data_id) {
        CacheType::Preference  // 用户偏好池（持久化）
    } else {
        CacheType::Temp        // 临时缓存池（LRU淘汰）
    };
    
    // 存入缓存
    cache_manager.store(&data_id, &data, cache_type, &source_id)?;
    
    Ok(data)
}
```

```javascript
// 前端错误处理
async function safeApiCall(fn, fallback = null) {
    try {
        return await fn();
    } catch (error) {
        console.error('API call failed:', error);
        
        // 根据错误类型处理
        if (error instanceof ApiError) {
            switch (error.code) {
                case ErrorCodes.NETWORK_ERROR:
                    showToast('网络连接失败，请检查网络设置');
                    break;
                case ErrorCodes.FILE_NOT_FOUND:
                    showToast('文件不存在，可能已被删除');
                    break;
                default:
                    showToast(`操作失败: ${error.message}`);
            }
        }
        
        return fallback;
    }
}
```
