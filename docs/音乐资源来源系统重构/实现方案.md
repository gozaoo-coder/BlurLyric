```rust
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]
pub enum Quality {
    Standard,    // 0 ~ 128 kbps
    Normal,      // 128 ~ 320 kbps
    HighQuality, // 320 ~ 900 kbps
    Lossless,    // 900 ~ 1500 kbps
    HiRes,       // > 1500 kbps
}

impl Quality {
    pub fn from_bitrate(bitrate: u32) -> Self {
        match bitrate {
            0..=128 => Quality::Standard,
            129..=320 => Quality::Normal,
            321..=900 => Quality::HighQuality,
            901..=1500 => Quality::Lossless,
            _ => Quality::HiRes,
        }
    }
}
```

#### merge_strategy.rs
```rust
use super::models::{Song, Album, Artist};

pub trait MergeStrategy: Send + Sync {
    fn is_same_song(&self, a: &Song, b: &Song) -> bool;
    fn is_same_album(&self, a: &Album, b: &Album) -> bool;
    fn is_same_artist(&self, a: &Artist, b: &Artist) -> bool;
}

pub struct StrictMergeStrategy;

impl MergeStrategy for StrictMergeStrategy {
    fn is_same_song(&self, a: &Song, b: &Song) -> bool {
        a.name == b.name && 
        a.artists.iter().map(|a| &a.name).collect::<Vec<_>>() == 
        b.artists.iter().map(|a| &a.name).collect::<Vec<_>>() &&
        a.album.as_ref().map(|al| &al.name) == b.album.as_ref().map(|al| &al.name)
    }

    fn is_same_album(&self, a: &Album, b: &Album) -> bool {
        a.name == b.name && 
        a.artists.iter().map(|a| &a.name).collect::<Vec<_>>() == 
        b.artists.iter().map(|a| &a.name).collect::<Vec<_>>()
    }

    fn is_same_artist(&self, a: &Artist, b: &Artist) -> bool {
        a.name == b.name
    }
}

pub struct FuzzyMergeStrategy;

impl MergeStrategy for FuzzyMergeStrategy {
    // 实现模糊匹配逻辑
    fn is_same_song(&self, a: &Song, b: &Song) -> bool {
        // 简化实现
        StrictMergeStrategy.is_same_song(a, b)
    }

    fn is_same_album(&self, a: &Album, b: &Album) -> bool {
        StrictMergeStrategy.is_same_album(a, b)
    }

    fn is_same_artist(&self, a: &Artist, b: &Artist) -> bool {
        StrictMergeStrategy.is_same_artist(a, b)
    }
}
```

### 阶段 2: 实现核心管理器

#### library_manager/music_storage_source_library_manager.rs
```rust
use std::collections::HashMap;
use std::sync::{Arc, Mutex};

use crate::core_v2::events::LibraryEvent;
use crate::core_v2::merge_strategy::{MergeStrategy, StrictMergeStrategy};
use crate::core_v2::models::{Song, Album, Artist, Trace, SongID, AlbumID, ArtistID};
use crate::sources::source::Source;

pub struct MusicStorageSourceLibraryManager {
    songs: Arc<Mutex<HashMap<SongID, Song>>>,
    albums: Arc<Mutex<HashMap<AlbumID, Album>>>,
    artists: Arc<Mutex<HashMap<ArtistID, Artist>>>,
    merge_strategy: Box<dyn MergeStrategy + Send + Sync>,
    event_handlers: Vec<Box<dyn Fn(&LibraryEvent) + Send + Sync>>,
}

impl MusicStorageSourceLibraryManager {
    pub fn new() -> Self {
        Self {
            songs: Arc::new(Mutex::new(HashMap::new())),
            albums: Arc::new(Mutex::new(HashMap::new())),
            artists: Arc::new(Mutex::new(HashMap::new())),
            merge_strategy: Box::new(StrictMergeStrategy),
            event_handlers: Vec::new(),
        }
    }

    pub fn register_song(&mut self, mut song: Song, trace: Trace) {
        let mut songs = self.songs.lock().unwrap();
        
        // 检查是否已存在相同歌曲
        for (existing_id, existing_song) in songs.iter() {
            if self.merge_strategy.is_same_song(&song, existing_song) {
                // 合并 - 追加 Trace
                let existing_song = songs.get_mut(existing_id).unwrap();
                existing_song.traces.push(trace.clone());
                self.emit_event(LibraryEvent::TraceAdded {
                    entity_id: existing_id.clone(),
                    trace,
                });
                return;
            }
        }
        
        // 新建实体
        song.traces.push(trace.clone());
        let song_id = song.id.clone();
        songs.insert(song_id.clone(), song);
        self.emit_event(LibraryEvent::TraceAdded {
            entity_id: song_id,
            trace,
        });
    }

    pub fn unregister_source_traces(&mut self, source_id: &str) {
        // 移除该来源的所有 Trace
    }

    pub fn get_song_file(&self, trace: &Trace) -> Option<Vec<u8>> {
        // 路由到对应 Source 获取
        None
    }

    fn emit_event(&self, event: LibraryEvent) {
        for handler in &self.event_handlers {
            handler(&event);
        }
    }
}

impl Default for MusicStorageSourceLibraryManager {
    fn default() -> Self {
        Self::new()
    }
}
```
